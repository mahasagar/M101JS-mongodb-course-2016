db.companies.aggregate([{
	$match: {
		"founded_year": 2004
		}
	
}, {
	$project: {
	        funding_rounds : 1,
		_id: 0,
		name : 1,
		count : 1,
		total_rounds: { $size: "$funding_rounds" },
		avg : {$avg :"$funding_rounds.raised_amount" }
	}
},{
	$match: {
		"total_rounds": {
			$gt: 4
		}
	}
},{
	$sort: {
		avg: 1
	}
},{ $limit: 1}]).pretty()
